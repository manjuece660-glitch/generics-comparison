Automatic Type Promotion in Java

üîë Rules

  1. Expression Promotion: byte, short, char ‚Üí int. If one operand is long ‚Üí long, if float ‚Üí float, if double ‚Üí double.
  2. Mixed Type Operations: smaller type is promoted to the larger type.
  3. Method Overloading: if no exact match, promotion happens step-by-step (byte ‚Üí short ‚Üí int ‚Üí long ‚Üí float ‚Üí double).
  4. Chained Promotions: promotion continues until a match is found.

üñ•Ô∏è Example Program

class PromotionDemo {
    // Overloaded methods
    void test(int x) { System.out.println("int method called: " + x); }
    void test(long x) { System.out.println("long method called: " + x); }
    void test(float x) { System.out.println("float method called: " + x); }
    void test(double x) { System.out.println("double method called: " + x); }

    public static void main(String[] args) {
        PromotionDemo demo = new PromotionDemo();

        // 1. Expression Promotion
        byte b1 = 10, b2 = 20;
        int sum = b1 + b2;
        System.out.println("byte + byte ‚Üí int: " + sum);

        char c = 'A'; // 65
        int charResult = c + 1;
        System.out.println("char + int ‚Üí int: " + charResult);

        short s1 = 5, s2 = 6;
        int shortResult = s1 * s2;
        System.out.println("short * short ‚Üí int: " + shortResult);

        // 2. Mixed Type
        int i = 10;
        long l = 20L;
        float f = 3.5f;
        double d = 2.7;

        System.out.println("int + long ‚Üí long: " + (i + l));
        System.out.println("long + float ‚Üí float: " + (l + f));
        System.out.println("float + double ‚Üí double: " + (f + d));

        // 3. Overloading
        demo.test(b1);
        demo.test(c);
        demo.test(s1);
        demo.test(i);
        demo.test(l);
        demo.test(f);
        demo.test(d);

        // 4. Chained Promotions
        demo.test(100);
        demo.test(100L);
        demo.test(100.0f);
        demo.test(100.0);
    }
}
